        }

            width: 50px;
            height: 50px;
            margin-top: 50px;
            background: pink;
        }

    </style>
</head>

    <!-- 相邻的垂直外边距塌陷，咱俩的关系没有那么好！ -->
    <div class="box1"></div>
    <div class="box2"></div>
    <!-- 嵌套的垂直外边距塌陷 儿子把爹边距下来了-->

        <div class="son">
        </div>
    </div>
